â­ If some operation is to be done in the array both from [0 -> n - 1] and [n - 1 -> 0]
	1. write code for [0 -> n - 1]: store in  ans1[]
	2. reverse the array
	3. copy code for [n - 1 -> 0], make changes to store the ans2[].
	4. reverse the ans2[]

Eg:

void solve()
{
    int n; cin >> n;
    vll v(n), pre(n), suff(n);
    cin >> v;

    f(i, 0, n - 1) {
        if(i == 0)  pre[i] = v[i];
        else pre[i] = pre[i - 1] + v[i];
    }

    reverse(all(v));  // trick
    
    f(i, 0, n - 1) {
        if(i == 0)  suff[i] = v[i];
        else suff[i] = suff[i - 1] + v[i];
    }

    reverse(all(suff));

    cout << pre << suff;
}
